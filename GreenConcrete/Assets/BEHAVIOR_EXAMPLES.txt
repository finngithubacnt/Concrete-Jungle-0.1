using UnityEngine;

public class ThrowableBehavior : MonoBehaviour, IItemBehavior
{
    public GameObject throwPrefab;
    public float throwForce = 10f;
    public int damage = 50;
    
    public void OnRightClick(InventoryItem item, GameObject player)
    {
        if (throwPrefab != null)
        {
            Vector3 spawnPos = player.transform.position + 
                              player.transform.forward * 1f;
            GameObject thrown = Instantiate(throwPrefab, spawnPos, 
                                           player.transform.rotation);
            
            Rigidbody rb = thrown.GetComponent<Rigidbody>();
            if (rb != null)
            {
                rb.AddForce(player.transform.forward * throwForce, 
                           ForceMode.Impulse);
            }
            
            Debug.Log($"Threw {item.definition.itemName}!");
            
            // Decrease stack
            item.currentStackSize--;
            if (item.currentStackSize <= 0)
            {
                PlayerInventory.Instance?.mainGrid?.RemoveItem(item);
            }
            PlayerInventory.Instance?.onInventoryChanged?.Invoke();
        }
    }
    
    public void OnUse(GameObject player) { OnRightClick(null, player); }
    public void OnEquip(GameObject player) { }
    public void OnUnequip(GameObject player) { }
    public bool CanUse(GameObject player) { return throwPrefab != null; }
}

â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”

Create Custom Script:

using UnityEngine;

public class ToolBehavior : MonoBehaviour, IItemBehavior
{
    public int maxDurability = 100;
    public int durabilityPerUse = 5;
    public GameObject toolModel;
    
    private int currentDurability;
    private bool isEquipped = false;
    
    void Awake()
    {
        currentDurability = maxDurability;
    }
    
    public void OnRightClick(InventoryItem item, GameObject player)
    {
        if (isEquipped)
        {
            OnUnequip(player);
        }
        else
        {
            OnEquip(player);
        }
    }
    
    public void OnUse(GameObject player)
    {
        currentDurability -= durabilityPerUse;
        
        if (currentDurability <= 0)
        {
            Debug.Log("Tool broke!");
            OnUnequip(player);
            // Remove from inventory
        }
        else
        {
            float percent = (currentDurability / (float)maxDurability) * 100;
            Debug.Log($"Tool durability: {percent:F0}%");
        }
    }
    
    public void OnEquip(GameObject player)
    {
        isEquipped = true;
        // Spawn tool model...
        Debug.Log("Tool equipped");
    }
    
    public void OnUnequip(GameObject player)
    {
        isEquipped = false;
        // Remove tool model...
        Debug.Log("Tool unequipped");
    }
    
    public bool CanUse(GameObject player)
    {
        return currentDurability > 0;
    }
}
