    // Decrease stack
    item.currentStackSize--;
    
    // Remove if empty
    if (item.currentStackSize <= 0)
    {
        InventoryGrid grid = PlayerInventory.Instance?.mainGrid;
        if (grid != null)
        {
            grid.RemoveItem(item);
        }
    }

    // Refresh UI
    PlayerInventory.Instance?.onInventoryChanged?.Invoke();
}
